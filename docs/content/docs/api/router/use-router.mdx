# `useRouter()`

The `useRouter` hook provides access to the router's internal state, including current path, parameters, and navigation methods.

## Hook Signature

```ts twoslash
// @errors: 2322
import { useRouter } from 'manicjs';

export default function Page() {
  const router = useRouter();
  // ...
}
```

## Examples

### Basic Usage

```tsx twoslash
// @errors: 2322
import React from 'react';
import { useRouter } from 'manicjs';

export default function UserProfile() {
  const { path, params } = useRouter();
  
  return (
    <div>
      <p>Current path: {path}</p>
      <p>User ID: {params.userId}</p>
    </div>
  );
}
```

### With Dynamic Path

```tsx twoslash
// @errors: 2322
import React from 'react';
import { Link } from 'manicjs';

export function PostList({ posts }: { posts: Array<{ slug: string; title: string }> }) {
  return (
    <ul>
      {posts.map(post => (
        <li key={post.slug}>
          <Link to={`/posts/${post.slug}`}>{post.title}</Link>
        </li>
      ))}
    </ul>
  );
}
```

### Multiple Route Parameters

```tsx twoslash
// @errors: 2322
import React from 'react';
import { useRouter } from 'manicjs';

// File: app/routes/users/[userId]/posts/[postId].tsx
export default function UserPost() {
  const { params } = useRouter();
  // { userId: "123", postId: "456" }
  
  return (
    <div>
      User {params.userId}'s Post {params.postId}
    </div>
  );
}
```

### Programmatic Navigation

```tsx twoslash
// @errors: 2322
import React from 'react';
import { useState } from 'react';
import { useRouter } from 'manicjs';

export function LoginButton() {
  const { navigate } = useRouter();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  
  const handleLogin = async () => {
    const res = await fetch('/api/login', {
      method: 'POST',
      body: JSON.stringify({ email, password }),
    });
    
    if (res.ok) {
      navigate('/dashboard');
    }
  };
  
  return <button onClick={handleLogin}>Login</button>;
}
```


### Route Guards

```tsx twoslash
// @errors: 2322
import React from 'react';
import { useEffect, useState } from 'react';
import { useRouter } from 'manicjs';

export default function AdminPage() {
  const { navigate, params } = useRouter();
  const [user, setUser] = useState<any>(null);
  
  useEffect(() => {
    // Fetch user data
    fetch('/api/me').then(res => res.json()).then(setUser);
  }, []);
  
  useEffect(() => {
    // Redirect if not admin
    if (user && !user.isAdmin) {
      navigate('/unauthorized', { replace: true });
    }
  }, [user]);

  if (!user?.isAdmin) return null;
  
  return <div>Admin Panel</div>;
}
```

## Type Definitions

```ts
interface NavigateOptions {
  replace?: boolean;
}

interface RouterContextValue {
  path: string;
  params: Record<string, string>;
  navigate: (to: string, options?: NavigateOptions) => void;
}

declare function useRouter(): RouterContextValue;
```

## Notes

- Must be called inside a component rendered by the router
- Returns `null` or throws if called outside router context
- `params` is empty object for static routes without dynamic segments
