# Theme

Manic provides built-in theme support for dark/light mode with system preference detection.

## Quick Start

### Wrap Your App

```tsx twoslash
// @errors: 2322
// app/main.tsx
import { ThemeProvider } from 'manicjs';

function App() {
  return <div>App</div>;
}

export default function Root() {
  return (
    <ThemeProvider>
      <App />
    </ThemeProvider>
  );
}
```

### Use in Component

```tsx twoslash
// @errors: 2322
// app/routes/index.tsx
import { useTheme, ThemeToggle } from 'manicjs';

export default function Home() {
  const { theme, isDark, toggle } = useTheme();
  
  return (
    <div className={isDark ? 'dark' : 'light'}>
      <p>Current: {theme}</p>
      <button onClick={toggle}>Toggle</button>
      <ThemeToggle />
    </div>
  );
}
```

## `useTheme()`

Access the current theme state and controls:

```ts
const { 
  theme,           // 'light' | 'dark' | 'system'
  setTheme,       // (theme: Theme) => void
  toggle,        // () => void
  isDark,        // boolean
  isLight,       // boolean
  resolvedTheme  // 'light' | 'dark' (resolved)
} = useTheme();
```

## `<ThemeToggle />`

A pre-built toggle button component:

```tsx twoslash
// @errors: 2322
import { ThemeToggle } from 'manicjs';
import { MdDarkMode, MdLightMode } from 'react-icons/md';

<>
  {/* Default */}
  <ThemeToggle />

  {/* With custom styling */}
  <ThemeToggle
    className="my-toggle"
    style={{ borderRadius: '8px' }}
  />

  {/* With custom icons (function children) */}
  <ThemeToggle>
    {(theme) =>
      theme === 'dark' ? (
        <MdDarkMode aria-hidden />
      ) : (
        <MdLightMode aria-hidden />
      )}
  </ThemeToggle>
</>
```

## CSS Variables

Manic sets CSS variables on the `html` element:

```css
:root {
  --manic-theme: light;
  --manic-is-dark: 0;
}

:root.dark {
  --manic-theme: dark;
  --manic-is-dark: 1;
}
```

### Using in Tailwind

```css
/* tailwind.config.js */
export default {
  darkMode: 'class',
  darkClass: 'dark',
}
```

```tsx twoslash
// @errors: 2322
// In components
const isDark = false;

<div className={isDark ? 'bg-gray-900 text-white' : 'bg-white text-gray-900'}>
  Content
</div>
```

## Programmatic Control

```tsx twoslash
// @errors: 2322
import { useTheme } from 'manicjs';
type Theme = 'light' | 'dark' | 'system';

function ThemeSwitch() {
  const { theme, setTheme } = useTheme();
  
  return (
    <select 
      value={theme}
      onChange={(e: any) => setTheme(e.target.value as Theme)}
    >
      <option value="light">Light</option>
      <option value="dark">Dark</option>
      <option value="system">System</option>
    </select>
  );
}
```

## Persistence

Theme preference is stored in `localStorage` under `manic-theme` key.
