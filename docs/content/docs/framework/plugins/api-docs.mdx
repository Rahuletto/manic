# API Docs

Generates interactive API reference documentation using Scalar.

## Installation

```bash
bun add @manicjs/api-docs
```

## Quick Start

### 1. Install Plugin

```bash
bun add @manicjs/api-docs
```

### 2. Configure

```ts
// manic.config.ts
import { defineConfig } from 'manicjs/config';
import { apiDocs } from '@manicjs/api-docs';

export default defineConfig({
  plugins: [
    apiDocs({
      route: '/api/docs',
    }),
  ],
});
```

### 3. Add OpenAPI Specs

The plugin generates docs from your API routes automatically.

## Auto-Generation

The plugin scans your API routes and generates OpenAPI spec:

```ts
// manic.config.ts
import { defineConfig } from 'manicjs/config';
import { apiDocs } from '@manicjs/api-docs';

export default defineConfig({
  plugins: [
    apiDocs({
      // Auto-generates from app/api/
      autoGenerate: true,
    }),
  ],
});
```

## Manual Spec

Provide custom OpenAPI spec:

```ts
import { defineConfig } from 'manicjs/config';
import { apiDocs } from '@manicjs/api-docs';

export default defineConfig({
  plugins: [
    apiDocs({
      spec: {
        openapi: '3.0.0',
        info: {
          title: 'My API',
          version: '1.0.0',
        },
        paths: {
          '/users': {
            get: {
              summary: 'List users',
              responses: {
                '200': {
                  description: 'OK',
                },
              },
            },
          },
        },
      },
    }),
  ],
});
```

## Themes

### Scalar (Default)

```ts
import { defineConfig } from 'manicjs/config';
import { apiDocs } from '@manicjs/api-docs';

export default defineConfig({
  plugins: [
    apiDocs({
      theme: 'scalar',
    }),
  ],
});
```

### OpenAPI

```ts
export default defineConfig({
  plugins: [
    apiDocs({
      theme: 'openapi',
    }),
  ],
});
```

## Authentication

Add auth to docs:

```ts
export default defineConfig({
  plugins: [
    apiDocs({
      route: '/api/docs',
      auth: {
        type: 'bearer',
        token: process.env.API_DOCS_TOKEN,
      },
    }),
  ],
});
```

## Custom Branding

```ts
export default defineConfig({
  plugins: [
    apiDocs({
      title: 'My App API',
      favicon: '/favicon.ico',
      colors: {
        primary: '#3b82f6',
      },
    }),
  ],
});
```

## Generated UI Features

- Interactive request/response playground
- Request body editor with JSON validation
- Authentication header support
- Code examples in multiple languages
- Search and filtering
- Dark/light mode

## Integration with OpenAPI

Export spec for other tools:

```ts
export default defineConfig({
  plugins: [
    apiDocs({
      output: './openapi.json',
      // Also generate spec file
      generateFile: true,
    }),
  ],
});
```

## Security

<Callout type="warn">

**Protect production docs** - Use authentication or disable in production.

</Callout>

```ts
export default defineConfig({
  plugins: [
    apiDocs({
      enabled: Bun.env.NODE_ENV !== 'production',
    }),
  ],
});
```
